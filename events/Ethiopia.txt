add_namespace = ethiopia

# Force Ethiopia to surrender
country_event = {
	id = ethiopia.1
	title = ethiopia.1.t
	desc = ethiopia.1.d
	picture = GFX_report_event_generic_army

	trigger = {
		tag = ETH
		has_war_with = ITA
		OR = {
			has_stability < 0.10
			surrender_progress > 0.75
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	fire_only_once = yes

	option = { # Accept demands
		name = ethiopia.1.a
		ai_chance = {
			factor = 100
		}

		ITA = {
			annex_country = { target = ETH transfer_troops = no }
			news_event = { id = ethiopia.2 days = 1 }
		}
		hidden_effect = {
			add_equipment_to_stockpile = { type = infantry_equipment_1 amount = 1500 producer = ITA }
			271 = {
				remove_core_of = ETH
			}
		}
		add_threat = -0.8
	}
}

news_event = {
	id = ethiopia.2
	title = ethiopia.2.t
	desc = ethiopia.2.d
	picture = GFX_report_event_generic_army

	is_triggered_only = yes

	fire_only_once = yes

	option = {
		name = ethiopia.2.a
	}
}
